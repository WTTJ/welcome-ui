# Refs

You can create a `ref`\* and pass it to a component to access the underlying DOM element.

Note: `ref`s can be created with either `React.createRef()` or `React.useRef(null)` for functional components or a callback ref if using a React `class`.

<Playground>
  {() => {
    const connectedInputRef = createRef()
    const normalInputRef = useRef(null)
    const nakedInputRef = useRef(null)
    const fileUploadRef = useRef(null)
    const buttonRef = useRef(null)
    //
    const handleConnectedClick = () => {
      connectedInputRef.current.focus()
    }
    const handleNormalClick = () => {
      normalInputRef.current.focus()
    }
    const handleNakedClick = () => {
      nakedInputRef.current.focus()
    }
    const handleFileUploadClick = () => {
      fileUploadRef.current.focus()
    }
    const handleButtonClick = () => {
      console.debug('<Button> ref', buttonRef)
    }
    //
    return (
      <Form
        onSubmit={() => {}}
        render={({ handleSubmit }) => (
          <form onSubmit={handleSubmit}>
            <ConnectedField
              name="firstName"
              component={InputText}
              label="Connected field"
              ref={connectedInputRef}
            />
            <Field
              name="lastName"
              component={InputText}
              label="Normal field"
              ref={normalInputRef}
            />
            <Label>Naked field</Label>
            <InputText name="naked" ref={nakedInputRef} />
            <Field
              name="avatar"
              component={FileUpload}
              label="Complex field"
              ref={fileUploadRef}
            />
            <p>Click a button below to focus the relevant field above</p>
            <Button onClick={handleConnectedClick}>Connected field</Button>
            <Button onClick={handleNormalClick} ml={20}>
              Normal field
            </Button>
            <Button onClick={handleNakedClick} ml={20}>
              Naked field
            </Button>
            <Button onClick={handleFileUploadClick} ml={20}>
              File upload field
            </Button>
            <p>Click the button below to read the button's ref in the console</p>
            <Button as="a" ref={buttonRef} onClick={handleButtonClick}>
              Test button
            </Button>
          </form>
        )}
      />
    )
  }}
</Playground>
