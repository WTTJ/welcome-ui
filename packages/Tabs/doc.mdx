# Tabs

<TagVersion version={version} name={name} />

Tabs make it easy to explore and switch between different views.

See [Reakit](https://reakit.io/docs/tab/) for documentation.

## Import

<Code>{`import { Tab, TabList, TabPanel } from '@welcome-ui/tabs'`}</Code>

## Default

Set the tab activated by default using `selectedId` on `useTabState`.

<Playground>
  {() => {
    const tab = useTabState({ selectedId: 'tab1' })
    return (
      <>
        <TabList aria-label="Tabs" {...tab}>
          <Tab {...tab} stopId="tab1">
            Tab 1
          </Tab>
          <Tab {...tab} stopId="tab2">
            Tab 2
          </Tab>
          <Tab {...tab} stopId="tab3">
            Tab 3
          </Tab>
          <Tab {...tab} stopId="tab4">
            Tab 4

            Tab 5
          </Tab>
        </TabList>
        <TabPanel {...tab} stopId="tab1">
          TabPanel 1

  }}
</Playground>

## With a badge

Add badges in tab.

<Playground>
  {() => {
    const tab = useTabState({ selectedId: 'tab2' })
    const getVariant = item => (tab.selectedId === item ? 'primary' : 'default')
    return (
      <>
        <TabList aria-label="Tabs" {...tab}>
          <Tab {...tab} stopId="tab1">
            Tab 1
            <Badge ml="xs" variant={getVariant('tab1')}>
              new
            </Badge>
          </Tab>
          <Tab {...tab} stopId="tab2">
            Tab 2
            <Badge ml="xs" variant={getVariant('tab2')}>
              old
            </Badge>
          </Tab>
          <Tab {...tab} stopId="tab3">
            Tab 3
          </Tab>
        </TabList>
        <TabPanel {...tab} stopId="tab1">
          TabPanel 1
        </TabPanel>
        <TabPanel {...tab} stopId="tab2">
          TabPanel 2
        </TabPanel>
        <TabPanel {...tab} stopId="tab3">
          TabPanel 3
        </TabPanel>
      </>
    )
  }}
</Playground>
