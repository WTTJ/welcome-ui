## Usage

```jsx
<Form
  initialValues={{}}
  validate={values => (!values.social ? { social: 'You must choose a social network' } : {})}
>
  <Field
    flexDirection="row"
    hint="Most popular is Twitter"
    label="Social networks"
    mb="xl"
    required
  >
    <RadioGroup name="social" options={constants.ITEMS} />
  </Field>
  <Button type="submit">Submit</Button>
</Form>
```

## With hint on radio

```jsx
<Form
  initialValues={{}}
  validate={values => (!values.social ? { social: 'You must choose a social network' } : {})}
>
  <Field flexDirection="row" mb="xl" required>
    <RadioGroup
      name="social"
      maxWidth={{ md: 0.3 }}
      options={[
        {
          value: 'behance',

        {
          value: 'dribbble',
          label: 'Dribbble',
          hint: 'Aenean a tristique turpis, in elementum est.',
        },
      ]}
    />
  </Field>
  <Button type="submit">Submit</Button>

```
