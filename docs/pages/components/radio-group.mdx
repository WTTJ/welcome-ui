## Usage

```jsx
function () {
  const [value, setValue] = React.useState('github')

  const handleChange = event => {
    setValue(event.target.value)
  }

  return (
    <Field
      label="Label"
      required
    >
      <RadioGroup name="social-1" options={constants.ITEMS.slice(0, 5)} onChange={handleChange} value={value} />
    </Field>
  )
}
```

## Row style

Add flexDirection to `row` to Field to add the inline style for radios group

```jsx
function () {
  const [value, setValue] = React.useState('github')

  const handleChange = event => {
    setValue(event.target.value)
  }

  return (
    <Field
      label="Label"
      required
      flexDirection="row"
    >
      <RadioGroup name="social-1" options={constants.ITEMS.slice(0, 5)} onChange={handleChange} value={value} />
    </Field>
  )
}
```

## Disabled

Add `disabled` to the Field component

```jsx
function () {
  return (
    <Field
      label="Label"
      disabled
    >
      <RadioGroup name="social-2" options={constants.ITEMS.slice(0, 5)} value="github" />
    </Field>
  )
}
```

## Variants

Use `warning`, `error`, `info` or `success` properties on Field to add a variant status on your RadioGroup. The label, hint or border color are set by magic ðŸª„

```jsx
function () {
  const ITEMS = constants.ITEMS.slice(0, 3)

  return (
    <Field
      label="Label"
      error="an error"
    >
      <RadioGroup name="social-3" options={ITEMS} />
    </Field>
  )
}
```

## With hint on radio

```jsx
function () {
  const [value, setValue] = React.useState(null)

  const handleChange = event => {
    setValue(event.target.value)
  }

  return (
    <Field flexDirection="row" mb="xl" required>
      <RadioGroup
        name="social-4"
        maxWidth={{ md: 0.3 }}
        options={[
          {
            value: 'behance',
            label: 'Behance',
            hint: 'Lorem ipsum dolor sit amet, consectetur adipiscing elit.',
          },
          {
            value: 'dribbble',
            label: 'Dribbble',
            hint: 'Aenean a tristique turpis, in elementum est.',
          },
        ]}
        onChange={handleChange}
        value={value}
      />
    </Field>
  )
}
```
