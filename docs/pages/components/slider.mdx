## Usage

```jsx
function () {
  const [value, setValue] = React.useState(null)

  const handleChange = (newValue) => {
    setValue(newValue)
  }

  return (
    <Stack spacing='xl'>
      <Slider min={0} max={100} value={50} onChange={handleChange} />
      <Slider.Range min={0} max={100} value={{min: 25, max: 75}} onChange={value => console.log(value)} />
    </Stack>
    )
}

```

## Label & Hint

`label` and `hint` props to add some information to your slider.

```jsx
<Stack>
  <Slider
    min={0}
    max={100}
    value={50}
    label="Label"
    hint="hint"
    onChange={value => console.log(value)}
  />
  <Slider.Range
    min={0}
    max={100}
    value={{ min: 20, max: 80 }}
    label="Label"
    hint="hint"
    onChange={value => console.log(value)}
  />
</Stack>
```

## Step

`step` prop allow you to snap to predefined sets of values.

```jsx
<Stack spacing="xl">
  <Slider min={0} max={100} value={10} step={10} onChange={value => console.log(value)} />
  <Slider.Range
    min={0}
    max={100}
    step={10}
    value={{ min: 20, max: 80 }}
    onChange={value => console.log(value)}
  />
</Stack>
```

## Types

There is 3 differents types for slider: `inline`, `left-field` and `right-field`.

```jsx
<Stack spacing="xl">
  <Slider min={0} max={100} value={50} onChange={value => console.log(value)} type="inline" />
  <Slider min={0} max={100} value={50} onChange={value => console.log(value)} type="left-field" />
  <Slider min={0} max={100} value={50} onChange={value => console.log(value)} type="right-field" />
</Stack>
```

There is only 2 differents types for range slider: `inline` and `fields`.

```jsx
<Stack spacing="xl">
  <Slider.Range
    value={{ min: 10, max: 90 }}
    min={0}
    max={100}
    onChange={value => console.log(value)}
    type="inline"
  />

  <Slider.Range
    value={{ min: 25, max: 75 }}
    min={0}
    max={100}
    onChange={value => console.log(value)}
    type="fields"
  />
</Stack>
```

## Values

You can specify a range of values that will add a mark for each of them.

```jsx
<Stack spacing="xl">
  <Slider
    min={0}
    max={100}
    value={50}
    onChange={value => console.log(value)}
    values={[0, 50, 100]}
  />

  <Slider
    min={0}
    max={100}
    value={25}
    onChange={value => console.log(value)}
    values={[0, 25, 50, 75, 100]}
  />

  <Slider.Range
    value={{ min: 40, max: 60 }}
    min={0}
    max={100}
    onChange={value => console.log(value)}
    values={[0, 50, 100]}
  />

  <Slider.Range
    value={{ min: 20, max: 80 }}
    min={0}
    max={100}
    onChange={value => console.log(value)}
    values={[0, 10, 20, 30, 40, 50, 60, 80, 70, 90, 100]}
  />
</Stack>
```

## borderSelectorColor

To match your background color, you should add a `borderSelectorColor`.

```jsx
<Stack backgroundColor="nude-100" spacing="xl">
  <Slider
    min={0}
    max={100}
    value={50}
    onChange={value => console.log(value)}
    borderSelectorColor="nude-100"
  />

  <Slider.Range
    min={0}
    max={100}
    value={{ min: 25, max: 75 }}
    onChange={value => console.log(value)}
  />
</Stack>
```
