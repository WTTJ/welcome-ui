## List of clickable tags

When we have a clickable tag, we apply an hover state

```jsx row
<Tag as="a" href="#" size="xs" onClick={() => console.log('clicked')}>
  Some link
</Tag>
<Tag as="a" href="https://www.welcome-ui.com/" target="_blank" size="sm">
  Welcome-ui
</Tag>
```

## OnClick

Simply add a function `onClick` to catch the action with an hover effect.

```jsx
function() {
  return (
    <Stack direction="row">
      <Tag onClick={() => console.log('clicked')}>Hover me</Tag>
      <Tag onClick={() => console.log('clicked')} variant="primary">Primary</Tag>
      <Tag onClick={() => console.log('clicked')} variant="secondary">Secondary</Tag>
      <Tag onClick={() => console.log('clicked')} variant="secondary-green">Hover me</Tag>
      <Tag onClick={() => console.log('clicked')} variant="secondary-blue">Hover me</Tag>
      <Tag onClick={() => console.log('clicked')} variant="secondary-red">Hover me</Tag>
      <Tag onClick={() => console.log('clicked')} variant="secondary-purple">Hover me</Tag>
    </Stack>
  )
}
```

## OnRemove

Simply add a function `onRemove` to reveal a cross icon with a remove action (or other).

```jsx
function() {
  const [isHide, setHide] = React.useState(false)

  return isHide ? (
    <Text fontSize="sm" fontWeight="bold" color="danger-50" m={0}>
      Tag removed
    </Text>
  ) : (
    <Stack>
      <Tag onRemove={() => setHide(true)}>Example of remove tag</Tag>
      <Tag size="sm" onRemove={() => setHide(true)}>small</Tag>
      <Tag variant="success" onRemove={() => setHide(true)}>success</Tag>
    </Stack>
  )
}
```
