## Examples

Use `checked` property

```jsx
function() {
  const [toggle, setToggle] = useState(false)
  const [toggleChecked, setToggleChecked] = useState(true)

  return (
    <>
      <Toggle mr="lg" checked={toggle} onClick={() => setToggle(!toggle)} />
      <Toggle checked={toggleChecked} onClick={() => setToggleChecked(!toggleChecked)} />
    </>
  )
}
```

## Field

Examples with `Field`

```jsx
function () {
  const [checked, setChecked] = useState(false)

  const handleToggle = () => {
    setChecked(!checked)
  }

  return (
    <Field label="Toggle">
      <Toggle mr="lg" checked={checked} onClick={handleToggle} disabled />
    </Field>
  )
}
```
