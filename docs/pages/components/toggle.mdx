## Examples

Use `checked` property

```jsx
function() {
  const [toggle, setToggle] = useState(false)
  const [toggleChecked, setToggleChecked] = useState(true)

  return (
    <>
      <Toggle mr="lg" checked={toggle} onClick={() => setToggle(!toggle)} />
      <Toggle checked={toggleChecked} onClick={() => setToggleChecked(!toggleChecked)} />
    </>
  )
}
```

## ConnectedField

Examples with `ConnectedField`

```jsx
<Form
  initialValues={{
    toggle: true,
    'toogle-checked': true,
    'toogle-disabled': true,
    'toggle-label': false,
    'disabled-label': true
  }}
>
  <Box display="flex" mb="lg">
    <ConnectedField name="toggle" component={Toggle} mr="lg" />
    <ConnectedField name="toggle-checked" component={Toggle} mr="lg" />
    <ConnectedField name="toggle-disabled" component={Toggle} disabled />
  </Box>
  <ConnectedField name="toggle-label" component={Toggle} label="Toggle with a label" mb="lg" />
  <ConnectedField name="disabled-label" component={Toggle} label="Disabled with a label" disabled />
</Form>
```
