## Basic

EmojiPicker is a field element that extends [Popover](https://www.welcome-ui.com/components/popover).  
The most basic emoji picker needs `useEmojiPicker()`, `<EmojiPicker.Trigger />` and `<EmojiPicker />`.

## Custom

```jsx
function Custom() {
  const emojiPicker = useEmojiPicker({ visible: true })
  const [emoji, setEmoji] = useState()

  return (
    <>
      <EmojiPicker.Trigger as={Button} {...emojiPicker}>
        {emoji ? emoji.alias : 'Open Picker'}
      </EmojiPicker.Trigger>
      <EmojiPicker onChange={setEmoji} {...emojiPicker}>
        <EmojiPicker.Tab name="Smileys">
          <EmojiPicker.List
            emojis={constants.EMOJIS.filter(emoji => emoji.category === 'Smileys & Emotion')}
          />
        </EmojiPicker.Tab>
        <EmojiPicker.Tab name="Animals">
          <EmojiPicker.List
            emojis={constants.EMOJIS.filter(emoji => emoji.category === 'Animals & Nature')}
          />
        </EmojiPicker.Tab>
      </EmojiPicker>
    </>
  )
}
```
