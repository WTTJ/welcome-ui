## With icons

```jsx
function () {
  const [value, setValue] = React.useState(null)

  const handleChange = event => {
    setValue(event.target.value)
  }

  return (
    <Field label="Icons" required>
      <Picker
        name="icon"
        options={[
          {
            element: ({ selected }) => (
              <Shape shape="circle" w={40} backgroundColor={selected ? 'nude-200' : 'unset'}>
                <EditIcon color="dark-900" />
              </Shape>
            ),
            value: 'edit',
          },
          {
            element: ({ selected }) => (
              <Shape shape="circle" w={40} backgroundColor={selected ? 'nude-200' : 'unset'}>
                <PencilIcon color="dark-900" />
              </Shape>
            ),
            value: 'pencil',
          },
        ]}
        onChange={handleChange}
        value={value}
      />
    </Field>
  )
}
```
