## About

See [Reakit](https://reakit.io/docs/tab/) for documentation.

## Usage

Set the tab activated by default using `selectedId` on `useTabState`.

```jsx
function() {
  const tabState = useTabState()

  return (
    <>
      <Tab.List aria-label="Tabs" state={tabState}>
        <Tab id="tab1">
          Tab 1
        </Tab>
        <Tab id="tab2">
          Tab 2
        </Tab>
        <Tab id="tab3">
          Tab 3
        </Tab>
        <Tab id="tab4">
          Tab 4
        </Tab>
        <Tab disabled id="tab5">
          Tab 5
        </Tab>
      </Tab.List>
      <Tab.Panel state={tabState} tabId="tab1">
        Tab.Panel 1
      </Tab.Panel>
      <Tab.Panel state={tabState} tabId="tab2">
        Tab.Panel 2
      </Tab.Panel>
      <Tab.Panel state={tabState} tabId="tab3">
        Tab.Panel 3
      </Tab.Panel>
      <Tab.Panel state={tabState} tabId="tab4">
        Tab.Panel 4
      </Tab.Panel>
      <Tab.Panel state={tabState} disabled tabId="tab5">
        Tab.Panel 5
      </Tab.Panel>
    </>
  )
}
```

## With a badge

Add badges, icons and images in tab.

```jsx
function() {
  const tabState = useTabState({ defaultSelectedId: 'tab2' })
  const getVariant = item => (tabState.selectedId === item ? 'primary' : 'default')

  return (
    <>
      <Tab.List aria-label="Tabs" state={tabState}>
        <Tab id="tab1">
          Tab 1
          <Badge variant={getVariant('tab1')}>
            new
          </Badge>
        </Tab>
        <Tab id="tab2">
          Tab 2
          <Badge variant={getVariant('tab2')}>
            old
          </Badge>
          <Badge shape="circle" variant={getVariant('tab2')}>
          2
          </Badge>
        </Tab>
        <Tab id="tab3">
          Tab 3
          <HeartIcon />
        </Tab>
        <Tab id="tab4">
          <img src="https://cdn.welcometothejungle.com/wttj-front/production/assets/images/logos/wttj.svg?v=3a3dbd7122a01600fb67e66b889bb47c" />
          Tab 4
        </Tab>
      </Tab.List>
      <Tab.Panel state={tabState} tabId="tab1">
        Tab.Panel 1
      </Tab.Panel>
      <Tab.Panel state={tabState} tabId="tab2">
        Tab.Panel 2
      </Tab.Panel>
      <Tab.Panel state={tabState} tabId="tab3">
        Tab.Panel 3
      </Tab.Panel>
      <Tab.Panel state={tabState} tabId="tab4">
        Tab.Panel 4
      </Tab.Panel>
```

```jsx
}
```

## One tab

Active bar doesn't display with only one tab.

```jsx
function() {
  const tabState = useTabState()

  return (
    <>
      <Tab.List aria-label="Tabs" state={tabState}>
        <Tab id="tab1">
          Tab 1
        </Tab>
      </Tab.List>
      <Tab.Panel state={tabState} tabId="tab1">
        Tab.Panel 1
      </Tab.Panel>
    </>
  )
}
```

## Vertical

Set the tab orientation by providing the `orientation` prop to the tab state

```jsx
function() {
  const tabState = useTabState({ orientation: 'vertical', defaultSelectedId: 'tab2' })

  return (
    <Box display="flex">
      <Tab.List w={200} mr="lg" aria-label="Tabs" state={tabState}>
        <Tab id="tab1">
          Tab 1
        </Tab>

          Tab 2
```
