## Usage

```jsx
function () {
  const  [value, setValue] = React.useState('behance')

  return (
    <Field label="Social networks" mb="xl" required>
      <RadioGroup name="social_1" options={constants.ITEMS.slice(0, 5)} renderOption={RadioTab} value={value} onChange={setValue} />
    </Field>
  )
}
```
