## Usage

```jsx

  const handleChange = event => {

  }

  return (
    <Field label="Social networks" mb="xl" required>
      <RadioGroup name="social_1" options={constants.ITEMS} renderOption={RadioTab} value={value} onChange={handleChange} />
    </Field>
  )
}
```
